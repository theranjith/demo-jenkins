    agent any

    stages {
        stage('Checkout') {
            steps {

            }
        }

            steps {
                echo 'Building project...'

            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
            }
        }

        stage('Deploy') {
            steps {
                echo 'Deploying app...'
            }
        }
    }
}
